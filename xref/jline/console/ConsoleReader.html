245         /*
246          * This is gross and here is how to fix it. In getCurrentPosition()

248          * tests and the fact that it is a unit test is determined by knowing
249          * if the original input stream was a ByteArrayInputStream. So, this
250          * is our test to do this.  What SHOULD happen is that the unit
251          * tests should pass in a terminal that is appropriately configured
252          * such that whatever behavior they expect to happen (or not happen)
253          * happens (or doesn't).
254          *
255          * So, TODO, get rid of this and fix the unit tests.
256          */
257         this.isUnitTestInput = in instanceof ByteArrayInputStream;
258         boolean nonBlockingEnabled =
259                escapeTimeout > 0L
260             && terminal.isSupported()
261             && in != null;
262 
263         /*
264          * If we had a non-blocking thread already going, then shut it down
265          * and start a new one.
266          */
267         if (this.in != null) {
268             this.in.shutdown();
269         }
270 
271         final InputStream wrapped = terminal.wrapInIfNeeded( in );
272 
273         this.in = new NonBlockingInputStream(wrapped, nonBlockingEnabled);
274         this.reader = new InputStreamReader( this.in, encoding );
275     }

475     private String stripAnsi(String str) {
476         if (str == null) return "";
477         try {
478             ByteArrayOutputStream baos = new ByteArrayOutputStream();
479             AnsiOutputStream aos = new AnsiOutputStream(baos);
480             aos.write(str.getBytes());
481             aos.flush();
482             return baos.toString();
483         } catch (IOException e) {
484             return str;
485         }

499     /**
500      * Set the current buffer's content to the specified {@link String}. The
501      * visual console will be modified to show the current buffer.
502      *
503      * @param buffer the new contents of the buffer.
504      */
505     private void setBuffer(final String buffer) throws IOException {
506         // don't bother modifying it if it is unchanged
507         if (buffer.equals(buf.buffer.toString())) {
508             return;
509         }
510 
511         // obtain the difference between the current buffer and the new one
512         int sameIndex = 0;
513 
514         for (int i = 0, l1 = buffer.length(), l2 = buf.buffer.length(); (i < l1)
515             && (i < l2); i++) {
516             if (buffer.charAt(i) == buf.buffer.charAt(i)) {
517                 sameIndex++;
518             }
519             else {
520                 break;
521             }
522         }
523 
524         int diff = buf.cursor - sameIndex;
525         if (diff < 0) { // we can't backspace here so try from the end of the buffer
526             moveToEnd();
527             diff = buf.buffer.length() - sameIndex;
528         }
529 
530         backspace(diff); // go back for the differences
531         killLine(); // clear to the end of the line
532         buf.buffer.setLength(sameIndex); // the new length
533         putString(buffer.substring(sameIndex)); // append the differences
534     }

567     /**
568      * Clear the buffer and add its contents to the history.
569      *
570      * @return the former contents of the buffer.
571      */
572     final String finishBuffer() throws IOException { // FIXME: Package protected because used by tests
573         String str = buf.buffer.toString();
574         String historyLine = str;
575 
576         if (expandEvents) {
577             str = expandEvents(str);
578             historyLine = str.replaceAll("\\!", "\\\\!");
579         }
580 
581         // we only add it to the history if the buffer is not empty
582         // and if mask is null, since having a mask typically means
583         // the string was a password. We clear the mask after this call
584         if (str.length() > 0) {
585             if (mask == null && isHistoryEnabled()) {
586                 history.add(historyLine);
587             }
588             else {
589                 mask = null;
590             }
591         }
592 
593         history.moveToEnd();
594 
595         buf.buffer.setLength(0);
596         buf.cursor = 0;
597 
598         return str;
599     }

601     /**
602      * Expand event designator such as !!, !#, !3, etc...
603      * See http://www.gnu.org/software/bash/manual/html_node/Event-Designators.html
604      */
605     protected String expandEvents(String str) throws IOException {
606         StringBuilder sb = new StringBuilder();
607         boolean escaped = false;
608         for (int i = 0; i < str.length(); i++) {
609             char c = str.charAt(i);
610             if (escaped) {
611                 sb.append(c);
612                 escaped = false;
613                 continue;
614             } else if (c == '\\') {
615                 escaped = true;
616                 continue;
617             } else {
618                 escaped = false;
619             }
620             switch (c) {
621                 case '!':
622                     if (i + 1 < str.length()) {
623                         c = str.charAt(++i);
624                         boolean neg = false;
625                         String rep = null;
626                         int i1, idx;
627                         switch (c) {
628                             case '!':
629                                 if (history.size() == 0) {
630                                     throw new IllegalArgumentException("!!: event not found");
631                                 }
632                                 rep = history.get(history.index() - 1).toString();
633                                 break;
634                             case '#':
635                                 sb.append(sb.toString());
636                                 break;
637                             case '?':
638                                 i1 = str.indexOf('?', i + 1);
639                                 if (i1 < 0) {
640                                     i1 = str.length();
641                                 }
642                                 String sc = str.substring(i + 1, i1);
643                                 i = i1;
644                                 idx = searchBackwards(sc);
645                                 if (idx < 0) {
646                                     throw new IllegalArgumentException("!?" + sc + ": event not found");
647                                 } else {
648                                     rep = history.get(idx).toString();
649                                 }
650                                 break;
651                             case ' ':
652                             case '\t':
653                                 sb.append('!');
654                                 sb.append(c);
655                                 break;
656                             case '-':
657                                 neg = true;
658                                 i++;
659                                 // fall through
660                             case '0':
661                             case '1':
662                             case '2':
663                             case '3':
664                             case '4':
665                             case '5':
666                             case '6':
667                             case '7':
668                             case '8':
669                             case '9':
670                                 i1 = i;
671                                 for (; i < str.length(); i++) {
672                                     c = str.charAt(i);
673                                     if (c < '0' || c > '9') {
674                                         break;
675                                     }
676                                 }
677                                 idx = 0;
678                                 try {
679                                     idx = Integer.parseInt(str.substring(i1, i));
680                                 } catch (NumberFormatException e) {
681                                     throw new IllegalArgumentException((neg ? "!-" : "!") + str.substring(i1, i) + ": event not found");
682                                 }
683                                 if (neg) {
684                                     if (idx < history.size()) {
685                                         rep = (history.get(history.index() - idx)).toString();
686                                     } else {
687                                         throw new IllegalArgumentException((neg ? "!-" : "!") + str.substring(i1, i) + ": event not found");
688                                     }
689                                 } else {
690                                     if (idx >= history.index() - history.size() && idx < history.index()) {
691                                         rep = (history.get(idx)).toString();
692                                     } else {
693                                         throw new IllegalArgumentException((neg ? "!-" : "!") + str.substring(i1, i) + ": event not found");
694                                     }
695                                 }
696                                 break;
697                             default:
698                                 String ss = str.substring(i);
699                                 i = str.length();
700                                 idx = searchBackwards(ss, history.index(), true);
701                                 if (idx < 0) {
702                                     throw new IllegalArgumentException("!" + ss + ": event not found");
703                                 } else {
704                                     rep = history.get(idx).toString();
705                                 }
706                                 break;
707                         }
708                         if (rep != null) {
709                             sb.append(rep);
710                         }
711                     } else {
712                         sb.append(c);
713                     }
714                     break;
715                 case '^':
716                     if (i == 0) {
717                         int i1 = str.indexOf('^', i + 1);
718                         int i2 = str.indexOf('^', i1 + 1);
719                         if (i2 < 0) {
720                             i2 = str.length();
721                         }
722                         if (i1 > 0 && i2 > 0) {
723                             String s1 = str.substring(i + 1, i1);
724                             String s2 = str.substring(i1 + 1, i2);
725                             String s = history.get(history.index() - 1).toString().replace(s1, s2);
726                             sb.append(s);
727                             i = i2 + 1;
728                             break;
729                         }
730                     }
731                     sb.append(c);
732                     break;
733                 default:
734                     sb.append(c);
735                     break;
736             }
737         }
738         if (escaped) {
739             sb.append('\\');
740         }
741         String result = sb.toString();
742         if (!str.equals(result)) {
743             print(result);
744             println();
745             flush();
746         }
747         return result;
748 
749     }

755         buf.write(str);
756         if (mask == null) {

767     /**
768      * Redraw the rest of the buffer from the cursor onwards. This is necessary
769      * for inserting text into the buffer.
770      *
771      * @param clear the number of characters to clear after the end of the buffer
772      */
773     private void drawBuffer(final int clear) throws IOException {
774         // debug ("drawBuffer: " + clear);
775         if (buf.cursor == buf.length() && clear == 0) {
776         } else {
777             char[] chars = buf.buffer.substring(buf.cursor).toCharArray();
778             if (mask != null) {
779                 Arrays.fill(chars, mask);
780             }
781             if (terminal.hasWeirdWrap()) {
782                 // need to determine if wrapping will occur:
783                 int width = terminal.getWidth();
784                 int pos = getCursorPosition();
785                 for (int i = 0; i < chars.length; i++) {
786                     print(chars[i]);
787                     if ((pos + i + 1) % width == 0) {
788                         print(32); // move cursor to next line by printing dummy space
789                         print(13); // CR / not newline.
790                     }
791                 }
792             } else {
793                 print(chars);
794             }
795             clearAhead(clear, chars.length);
796             if (terminal.isAnsiSupported()) {
797                 if (chars.length > 0) {
798                     back(chars.length);
799                 }
800             } else {
801                 back(chars.length);

1222     /**
1223      * Searches forward of the current position for a character and moves
1224      * the cursor onto it.
1225      * @param count Number of times to repeat the process.
1226      * @param ch The character to search for
1227      * @return true if the char was found, false otherwise
1228      * @throws IOException
1229      */
1230     private boolean viCharSearch(int count, int invokeChar, int ch) throws IOException {
1231         if (ch < 0 || invokeChar < 0) {
1232             return false;
1233         }
1234 
1235         char    searchChar = (char)ch;
1236         boolean isForward;
1237         boolean stopBefore;
1238 
1239         /*
1240          * The character stuff turns out to be hairy. Here is how it works:
1241          *   f - search forward for ch
1242          *   F - search backward for ch
1243          *   t - search forward for ch, but stop just before the match
1244          *   T - search backward for ch, but stop just after the match
1245          *   ; - After [fFtT;], repeat the last search, after ',' reverse it
1246          *   , - After [fFtT;], reverse the last search, after ',' repeat it
1247          */
1248         if (invokeChar == ';' || invokeChar == ',') {
1249             // No recent search done? Then bail
1250             if (charSearchChar == 0) {
1251                 return false;
1252             }
1253 
1254             // Reverse direction if switching between ',' and ';'
1255             if (charSearchLastInvokeChar == ';' || charSearchLastInvokeChar == ',') {
1256                 if (charSearchLastInvokeChar != invokeChar) {
1257                     charSearchFirstInvokeChar = switchCase(charSearchFirstInvokeChar);
1258                 }
1259             }
1260             else {
1261                 if (invokeChar == ',') {
1262                     charSearchFirstInvokeChar = switchCase(charSearchFirstInvokeChar);
1263                 }
1264             }
1265 
1266             searchChar = charSearchChar;
1267         }
1268         else {
1269             charSearchChar            = searchChar;
1270             charSearchFirstInvokeChar = (char) invokeChar;
1271         }
1272 
1273         charSearchLastInvokeChar = (char)invokeChar;
1274 
1275         isForward = Character.isLowerCase(charSearchFirstInvokeChar);
1276         stopBefore = (Character.toLowerCase(charSearchFirstInvokeChar) == 't');
1277 
1278         boolean ok = false;
1279 
1280         if (isForward) {
1281             while (count-- > 0) {
1282                 int pos = buf.cursor + 1;
1283                 while (pos < buf.buffer.length()) {
1284                     if (buf.buffer.charAt(pos) == (char) searchChar) {
1285                         setCursorPosition(pos);
1286                         ok = true;
1287                         break;
1288                     }
1289                     ++pos;
1290                 }
1291             }
1292 
1293             if (ok) {
1294                 if (stopBefore)
1295                     moveCursor(-1);
1296 
1297                 /*
1298                  * When in yank-to, move-to, del-to state we actually want to
1299                  * go to the character after the one we landed on to make sure
1300                  * that the character we ended up on is included in the
1301                  * operation
1302                  */
1303                 if (isInViMoveOperationState()) {
1304                     moveCursor(1);
1305                 }
1306             }
1307         }
1308         else {
1309             while (count-- > 0) {
1310                 int pos = buf.cursor - 1;
1311                 while (pos >= 0) {
1312                     if (buf.buffer.charAt(pos) == (char) searchChar) {
1313                         setCursorPosition(pos);
1314                         ok = true;
1315                         break;
1316                     }
1317                     --pos;
1318                 }
1319             }
1320 
1321             if (ok && stopBefore)
1322                 moveCursor(1);
1323         }
1324 
1325         return ok;
1326     }

1572          * so we discard it.
1573          */

1592             }

1597                     idx = i;

1626          * as movement, any other key is treated as if you are editing the
1627          * line with it, so we return it back up to the caller for interpretation.
1628          */

1631             boolean forward = isForward;
1632             switch (ch) {
1633                 case 'p': case 'P':
1634                     forward = !isForward;
1635                     // Fallthru
1636                 case 'n': case 'N':
1637                     boolean isMatch = false;
1638                     if (forward) {
1639                         for (int i = idx+1; !isMatch && i < end; i++) {
1640                             if (history.get(i).toString().contains(searchTerm)) {
1641                                 idx = i;
1642                                 isMatch = true;
1643                             }
1644                         }
1645                     }
1646                     else {
1647                         for (int i = idx - 1; !isMatch && i >= start; i--) {
1648                             if (history.get(i).toString().contains(searchTerm)) {
1649                                 idx = i;
1650                                 isMatch = true;
1651                             }
1652                         }
1653                     }
1654                     if (isMatch) {
1655                         setCursorPosition(0);
1656                         killLine();
1657                         putString(history.get(idx));
1658                         setCursorPosition(0);
1659                     }
1660                     break;
1661                 default:
1662                     isComplete = true;
1663             }
1664             flush();
1665         }
1666 
1667         /*
1668          * Complete?
1669          */
1670         return ch;

1702
